Vanilla Java

01 class ConditionExample {
02   private volatile boolean paused;
03   private final java.util.concurrent.locks.Lock $canResumeLock = new java.util.concurrent.locks.ReentrantLock();
04   private final java.util.concurrent.locks.Condition canResume = $canResumeLock.newCondition();
05   
06   @java.lang.SuppressWarnings("all")
07   void unpause() {
08     this.$canResumeLock.lock();
09     try {
10       paused = false;
11       this.canResume.signal();
12     } finally {
13       this.$canResumeLock.unlock();
14     }
15   }
16   
17   @java.lang.SuppressWarnings("all")
18   void pause() throws java.lang.InterruptedException {
19     this.$canResumeLock.lock();
20     try {
21       while (this.isPaused()) {
22         this.canResume.await();
23       }
24     } finally {
25       this.$canResumeLock.unlock();
26     }
27   }
28   
29   @java.lang.SuppressWarnings("all")
30   private boolean isPaused() {
31     return this.paused;
32   }
33 }
